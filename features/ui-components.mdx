SameVibe uses a modern UI component system built on Radix UI primitives, styled with Tailwind CSS, featuring glassmorphism effects and dark mode support.

## Component Architecture

The UI system consists of three layers:

1. **Radix UI Primitives**: Unstyled, accessible component primitives
2. **Base Components**: Styled wrappers in `components/ui/`
3. **Custom Components**: Application-specific components in `components/`

## Base UI Components

Base components are located in `components/ui/` and provide styled implementations of common UI elements.

### Button Component

The Button component uses `class-variance-authority` for variant management.

<CodeGroup>
```typescript components/ui/button.tsx
import * as React from "react"
import { Slot } from "@radix-ui/react-slot"
import { cva, type VariantProps } from "class-variance-authority"
import { cn } from "@/lib/utils"

const buttonVariants = cva(
  "inline-flex items-center justify-center gap-2 whitespace-nowrap rounded-md text-sm font-medium transition-all disabled:pointer-events-none disabled:opacity-50",
  {
    variants: {
      variant: {
        default: "bg-primary text-primary-foreground hover:bg-primary/90",
        destructive: "bg-destructive text-white hover:bg-destructive/90",
        outline: "border bg-background shadow-xs hover:bg-accent",
        secondary: "bg-secondary text-secondary-foreground hover:bg-secondary/80",
        ghost: "hover:bg-accent hover:text-accent-foreground",
        link: "text-primary underline-offset-4 hover:underline",
      },
      size: {
        default: "h-9 px-4 py-2",
        sm: "h-8 rounded-md gap-1.5 px-3",
        lg: "h-10 rounded-md px-6",
        icon: "size-9",
        "icon-sm": "size-8",
        "icon-lg": "size-10",
      },
    },
    defaultVariants: {
      variant: "default",
      size: "default",
    },
  }
)

function Button({
  className,
  variant,
  size,
  asChild = false,
  ...props
}: React.ComponentProps<"button"> &
  VariantProps<typeof buttonVariants> & {
    asChild?: boolean
  }) {
  const Comp = asChild ? Slot : "button"

  return (
    <Comp
      className={cn(buttonVariants({ variant, size, className }))}
      {...props}
    />
  )
}

export { Button, buttonVariants }
```
</CodeGroup>

### Card Component

The Card component provides a container with consistent styling.

<CodeGroup>
```typescript components/ui/card.tsx
import * as React from "react"
import { cn } from "@/lib/utils"

function Card({ className, ...props }: React.ComponentProps<"div">) {
  return (
    <div
      className={cn(
        "bg-card text-card-foreground flex flex-col gap-6 rounded-xl border py-6 shadow-sm",
        className
      )}
      {...props}
    />
  )
}

function CardHeader({ className, ...props }: React.ComponentProps<"div">) {
  return (
    <div
      className={cn(
        "@container/card-header grid auto-rows-min items-start gap-2 px-6",
        className
      )}
      {...props}
    />
  )
}

function CardTitle({ className, ...props }: React.ComponentProps<"div">) {
  return (
    <div
      className={cn("leading-none font-semibold", className)}
      {...props}
    />
  )
}

function CardContent({ className, ...props }: React.ComponentProps<"div">) {
  return (
    <div
      className={cn("px-6", className)}
      {...props}
    />
  )
}

export { Card, CardHeader, CardTitle, CardContent }
```
</CodeGroup>

#### Card Sub-components

- `Card`: Main container with rounded borders and shadow
- `CardHeader`: Header section with grid layout
- `CardTitle`: Title with semibold font
- `CardContent`: Content area with horizontal padding
- `CardFooter`: Footer section for actions
- `CardDescription`: Muted text for descriptions
- `CardAction`: Right-aligned action slot

### Custom Animated Components

SameVibe includes custom animated components for enhanced visual effects.

#### AuroraText

Animated gradient text with color cycling.

<CodeGroup>
```typescript components/ui/aurora-text.tsx
"use client"

import React, { memo } from "react"

interface AuroraTextProps {
  children: React.ReactNode
  className?: string
  colors?: string[]
  speed?: number
}

export const AuroraText = memo(
  ({
    children,
    className = "",
    colors = ["#FF0080", "#7928CA", "#0070F3", "#38bdf8"],
    speed = 1,
  }: AuroraTextProps) => {
    const gradientStyle = {
      backgroundImage: `linear-gradient(135deg, ${colors.join(", ")}, ${
        colors[0]
      })`,
      WebkitBackgroundClip: "text",
      WebkitTextFillColor: "transparent",
      animationDuration: `${10 / speed}s`,
    }

    return (
      <span className={`relative inline-block ${className}`}>
        <span className="sr-only">{children}</span>
        <span
          className="animate-aurora relative bg-[length:200%_auto] bg-clip-text text-transparent"
          style={gradientStyle}
        >
          {children}
        </span>
      </span>
    )
  }
)
```
</CodeGroup>

**Props:**
- `colors`: Array of color values for gradient
- `speed`: Animation speed multiplier (default: 1)

#### ShimmerButton

Button with animated shimmer effect.

<CodeGroup>
```typescript components/ui/shimmer-button.tsx
export interface ShimmerButtonProps extends ComponentPropsWithoutRef<"button"> {
  shimmerColor?: string
  shimmerSize?: string
  borderRadius?: string
  shimmerDuration?: string
  background?: string
}

export const ShimmerButton = React.forwardRef<HTMLButtonElement, ShimmerButtonProps>(
  ({
    shimmerColor = "#ffffff",
    shimmerSize = "0.05em",
    shimmerDuration = "3s",
    borderRadius = "100px",
    background = "rgba(0, 0, 0, 1)",
    className,
    children,
    ...props
  }, ref) => {
    return (
      <button
        style={{
          "--shimmer-color": shimmerColor,
          "--radius": borderRadius,
          "--speed": shimmerDuration,
          "--bg": background,
        } as CSSProperties}
        className={cn(
          "group relative z-0 flex cursor-pointer items-center justify-center overflow-hidden [border-radius:var(--radius)] border border-white/10 px-6 py-3 whitespace-nowrap text-white [background:var(--bg)]",
          className
        )}
        ref={ref}
        {...props}
      >
        {children}
      </button>
    )
  }
)
```
</CodeGroup>

**Props:**
- `shimmerColor`: Color of the shimmer effect
- `shimmerDuration`: Animation duration
- `borderRadius`: Button border radius
- `background`: Button background color

### Glassmorphism Effects

Glasmorphism is achieved using backdrop filters:

```css
/* Navigation bar */
backdrop-blur
bg-background/95
supports-[backdrop-filter]:bg-background/60

/* ProfileAside card */
backdrop-blur-md
bg-background/80
```

**Key Classes:**
- `backdrop-blur`: Standard blur effect
- `backdrop-blur-md`: Medium blur effect
- `bg-{color}/{opacity}`: Transparent background
- `supports-[backdrop-filter]:` Feature query for browser support

### Dark Mode

The application uses a dark theme by default with custom OKLCH colors:

```css
:root {
  --background: oklch(0.147 0.004 49.25); /* Dark background */
  --foreground: oklch(0.985 0.001 106.423); /* Light text */
}
```

Dark mode is implemented at the CSS variable level using the `@custom-variant dark` directive:

```css
@custom-variant dark (&:is(.dark *));
```

## Utility Function

The `cn` utility function combines Tailwind classes:

```typescript lib/utils.ts
import { clsx, type ClassValue } from "clsx"
import { twMerge } from "tailwind-merge"

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}
```

**Usage:**
```typescript
className={cn(
  "base-classes",
  variant === "primary" && "primary-classes",
  className // User-provided classes
)}
```

The `cn` function:
1. Uses `clsx` for conditional class names
2. Uses `twMerge` to resolve conflicting Tailwind classes

## Component Patterns

<AccordionGroup>
  <Accordion title="Polymorphic Components">
    Components like Button use the `asChild` prop with Radix Slot for polymorphism:
    
    ```tsx
    <Button asChild>
    ```

  <Accordion title="Client Components">
    Components with interactivity use `"use client"` directive:
    
    ```tsx
    "use client";
    
    export function InteractiveComponent() {
      const [state, setState] = useState();
      // ...
    }
    ```
  </Accordion>
  
  <Accordion title="Responsive Design">
    Components use Tailwind responsive prefixes:
    
    ```tsx
    className="hidden 2xl:block" // Desktop only
    className="2xl:hidden" // Mobile/tablet only
    ```
  </Accordion>
  
  <Accordion title="Server Components">
    Components without client-side logic are server components by default, improving performance.
  </Accordion>
